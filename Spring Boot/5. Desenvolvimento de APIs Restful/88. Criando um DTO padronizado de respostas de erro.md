
public record ErroCampo(String campo, String erro) {  
}

public record ErroResposta (int status, String mensagem, List<ErroCampo> erro) {  
  
    public static ErroResposta respostaPadrao(String mensagem){  
        return new ErroResposta(HttpStatus.BAD_REQUEST.value(), mensagem, List.of());  
    }  
  
    public static ErroResposta conflito(String mensagem){  
        return new ErroResposta(HttpStatus.CONFLICT.value(), mensagem, List.of());  
    }  
  
}

public class RegistroDuplicado extends RuntimeException{  
    public RegistroDuplicado(String message) {  
        super(message);  
    }  
}

@Component  
public class AutorValidator {  
    @Autowired  
    private AutorRepository autorRepository;  
  
    public AutorValidator(AutorRepository autorRepository) {  
        this.autorRepository = autorRepository;  
    }  
}

@Autowired  
private AutorRepository autorRepository;  
private AutorService autorService;  
  
public AutorService(AutorRepository autorRepository, AutorService autorService) {  
    this.autorRepository = autorRepository;  
    this.autorService = autorService;  
}  
  
public Autor salvar(Autor autor){  
    return autorRepository.save(autor);  
}  
  
public void atualizar(Autor autor){  
    if (autor.getId() == null){  
        throw new IllegalArgumentException("Para atualizarr é necessário que esteja salvo");  
    } else {  
        autorRepository.save(autor);  
    }  
}
